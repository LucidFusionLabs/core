set(LFL_GLES2 1)
add_definitions(${LFL_APP_DEF})
include_directories(${LFL_APP_INCLUDE})
lfl_add_library(app_null_graphics STATIC null_graphics.cpp)

lfl_add_library(app_opengl_graphics STATIC opengl_graphics.cpp)
target_include_directories(app_opengl_graphics PUBLIC ${OPENGL_INCLUDE})
target_link_libraries(app_opengl_graphics PUBLIC ${OPENGL_LIB})
if(LFL_GLES1)
  target_compile_definitions(app_opengl_graphics PUBLIC -DLFL_GLES1)
endif()
if(LFL_GLES2)
  target_compile_definitions(app_opengl_graphics PUBLIC -DLFL_GLES2)
endif()
if(LFL_GLEW)
  target_compile_definitions(app_opengl_graphics PUBLIC -DLFL_GLEW)
  target_include_directories(app_opengl_graphics PUBLIC ${GLEW_INCLUDE})
  target_link_libraries(app_opengl_graphics PUBLIC ${GLEW_LIB})
endif()

if(LFL_QT)
  lfl_enable_qt()
  add_library(app_qt_graphics STATIC opengl_graphics.cpp)
  target_compile_definitions(app_qt_graphics PUBLIC -DLFL_QTGL)
  target_include_directories(app_qt_graphics PUBLIC ${QT_INCLUDE})
  if(LFL_GLES1)
    target_compile_definitions(app_qt_graphics PUBLIC -DLFL_GLES1)
  endif()
  if(LFL_GLES2)
    target_compile_definitions(app_qt_graphics PUBLIC -DLFL_GLES2)
  endif()
  target_link_libraries(app_qt_graphics PUBLIC ${OPENGL_LIB})
endif()
